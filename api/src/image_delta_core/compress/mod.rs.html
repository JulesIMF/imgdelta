47/// Run the full 8-stage compress pipeline for one Fs partition.
48///
49/// Drop-in replacement for `compress_pipeline::compress_fs_partition`.
50#[allow(clippy::too_many_arguments)]
51pub async fn compress_fs_partition(
52    base_root: &Path,
53    target_root: &Path,
54    descriptor: &PartitionDescriptor,
55    storage: Arc<dyn Storage>,
56    image_id: &str,
57    base_image_id: Option<&str>,
58    router: Arc<RouterEncoder>,
59    fs_type: &str,
60    workers: usize,
61    debug_dir: Option<&Path>,
62) -> Result<(PartitionManifest, bool, u64)> {
63    let tmp_dir = tempfile::TempDir::new()?;
64
65    // Stage 1: walkdir (needs filesystem paths, called outside the pipeline).
66    let draft = walkdir_fn(base_root, target_root)?;
67
68    // Stages 2–7 via the pipeline runner.
69    let ctx = StageContext {
70        storage: Arc::clone(&storage),
71        router,
72        image_id: Arc::from(image_id),
73        base_image_id: base_image_id.map(Arc::from),
74        partition_number: Some(descriptor.number as i32),
75        workers,
76        tmp_dir: Arc::from(tmp_dir.path()),
77        debug_dir: debug_dir.map(Arc::from),
78    };
79
80    let pipeline = CompressPipeline::default_fs();
81    let draft = pipeline.run(&ctx, draft, debug_dir).await?;
82
83    // Stage 8: pack and upload archive.
84    let (content, patches_compressed, archive_stored_bytes) =
85        pack_and_upload_archive_fn(draft, storage.as_ref(), image_id, fs_type).await?;
86
87    Ok((
88        PartitionManifest {
89            descriptor: descriptor.clone(),
90            content,
91        },
92        patches_compressed,
93        archive_stored_bytes,
94    ))
95}
