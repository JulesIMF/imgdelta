19// ── Stage struct ──────────────────────────────────────────────────────────────

21/// Stage 2: for each newly added regular file (`old_path = None, data = LazyBlob`),
22/// find a matching blob in `base_image_id` via path similarity.
23///
24/// Files that find a match are upgraded to delta candidates:
25/// `data = BlobRef(uuid)` and `patch = Lazy { old: BlobRef(uuid), new: FilePath(...) }`.
26/// The patch will be computed in stage 7 with the blob as the delta base.
27pub struct S3Lookup;
28
29#[async_trait]
30impl CompressStage for S3Lookup {
31    fn name(&self) -> &'static str {
32        "s3_lookup"
33    }
34
35    async fn run(&self, ctx: &StageContext, draft: FsDraft) -> Result<FsDraft> {
36        let base_image_id = match &ctx.base_image_id {
37            Some(id) => id.as_ref(),
38            None => return Ok(draft),
39        };
40        s3_lookup_fn(
41            draft,
42            ctx.storage.as_ref(),
43            base_image_id,
44            ctx.partition_number,
45        )
46        .await
47    }
48}
49
50// ── Implementation ────────────────────────────────────────────────────────────
51
52pub async fn s3_lookup_fn(
53    mut draft: FsDraft,
54    storage: &dyn crate::storage::Storage,
55    base_image_id: &str,
56    partition_number: Option<i32>,
57) -> Result<FsDraft> {
58    let candidates = storage
59        .find_blob_candidates(base_image_id, partition_number)
60        .await?;
61    if candidates.is_empty() {
62        debug!("s3_lookup: no candidates in storage for base_image_id={base_image_id}");
63        return Ok(draft);
64    }
65    debug!(
66        base_image_id,
67        candidates = candidates.len(),
68        "s3_lookup: candidate blobs available"
69    );
70
71    let target_paths: Vec<String> = draft
72        .records
73        .iter()
74        .filter(|r| {
75            r.old_path.is_none()
76                && r.entry_type == EntryType::File
77                && matches!(r.data, Some(Data::LazyBlob(_)))
78        })
79        .filter_map(|r| r.new_path.clone())
80        .collect();
81
82    if target_paths.is_empty() {
83        return Ok(draft);
84    }
85
86    let source_paths: Vec<String> = candidates.iter().map(|c| c.original_path.clone()).collect();
87    let matches = find_best_matches(&source_paths, &target_paths, &PathMatchConfig::default())?;
88
89    let match_map: HashMap<&str, uuid::Uuid> = matches
90        .iter()
91        .filter_map(|m| {
92            candidates
93                .iter()
94                .find(|c| c.original_path == m.source_path)
95                .map(|c| (m.target_path.as_str(), c.uuid))
96        })
97        .collect();
98
99    for record in &mut draft.records {
100        let new_path = match &record.new_path {
101            Some(p) => p.clone(),
102            None => continue,
103        };
104        if let Some(&blob_id) = match_map.get(new_path.as_str()) {
105            if let Some(Data::LazyBlob(lazy_path)) = &record.data {
106                let lazy_path = lazy_path.clone();
107                let blob_ref = BlobRef {
108                    blob_id,
109                    size: record.size,
110                };
111                debug!(
112                    path = %new_path,
113                    blob_id = %blob_id,
114                    "s3_lookup: matched to base blob"
115                );
116                record.patch = Some(Patch::Lazy {
117                    old_data: DataRef::BlobRef(blob_ref.clone()),
118                    new_data: DataRef::FilePath(lazy_path),
119                });
120                record.data = Some(Data::BlobRef(blob_ref));
121            }
122        }
123    }
124
125    Ok(draft)
126}
