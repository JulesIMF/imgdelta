26/// Handles compression for a single partition, regardless of type.
27///
28/// Implementations exist for:
29/// - [`FsPartitionCompressor`] — mounts and runs the 8-stage pipeline, writing
30///   patch files into `ctx.patches_dir`.
31/// - [`BiosBootCompressor`] — reads raw bytes and uploads as a single blob.
32/// - [`RawPartitionCompressor`] — reads raw bytes and uploads as a single blob.
33/// - [`MbrCompressor`] — reads 440 MBR bytes and uploads as a single blob.
34///
35/// Patch bytes are never returned in memory.  FS compressors write individual
36/// patch files to `ctx.patches_dir`; the orchestrator packs them into one
37/// archive after all partitions are done.
38#[async_trait]
39pub trait PartitionCompressor: Send + Sync {
40    /// Compress one partition and return its manifest.
41    /// The second element is the number of bytes actually written to blob storage
42    /// (deduped blobs already in storage are not counted).
43    /// For Fs partitions this is always 0 — their blobs are tracked inside
44    /// [`PartitionContent::Fs::blobs_stored_bytes`] and summed by the orchestrator.
45    async fn compress(
46        &self,
47        ctx: &CompressContext,
48        handle: PartitionHandle,
49    ) -> Result<(PartitionManifest, u64)>;
50}
