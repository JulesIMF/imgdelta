4// image-delta — incremental disk-image compression toolkit
5// Unsafe FFI bindings to the libxdelta3 C library
6
7// All `unsafe` code in image-delta-core is confined to this file.
8// The safe wrapper in `mod.rs` is the only public interface to the outside.
9//
10// xdelta3 single-file C library:
11//   xd3_encode_memory(input, input_size, source, source_size,
12//                     output_buf, output_size, avail_output, flags) -> int
13//   xd3_decode_memory(input, input_size, source, source_size,
14//                     output_buf, output_size, avail_output, flags) -> int
15//
16// `usize_t` in xdelta3.h resolves to `uint64_t` because XD3_USE_LARGESIZET=1
17// is defined unconditionally in the header when targeting 64-bit platforms.
18// We use `u64` for all `usize_t` parameters.
19//
20// Return value 0 = success; non-zero = errno (ENOSPC = 28 when the output
21// buffer is too small).
22
23use std::os::raw::c_int;
24
25/// C `usize_t` as seen by xdelta3 on 64-bit targets.
26pub type UsiZeT = u64;
27
28/// Standard POSIX `ENOSPC` — returned by xdelta3 when the output buffer is
29/// too small. Value 28 is stable on Linux/glibc and musl.
30pub const ENOSPC: c_int = 28;
31
32unsafe extern "C" {
33    /// Encode `input` (new/target content) against `source` (base content)
34    /// and write the VCDIFF delta to `output_buffer`.
35    ///
36    /// Returns 0 on success, `ENOSPC` when `avail_output` is insufficient,
37    /// or another errno value on other errors.
38    pub fn xd3_encode_memory(
39        input: *const u8,
40        input_size: UsiZeT,
41        source: *const u8,
42        source_size: UsiZeT,
43        output_buffer: *mut u8,
44        output_size: *mut UsiZeT,
45        avail_output: UsiZeT,
46        flags: c_int,
47    ) -> c_int;
48
49    /// Decode `input` (VCDIFF delta bytes) against `source` (base content)
50    /// and write the reconstructed target to `output_buf`.
51    ///
52    /// Returns 0 on success, `ENOSPC` when `avail_output` is insufficient,
53    /// or another errno value on other errors.
54    pub fn xd3_decode_memory(
55        input: *const u8,
56        input_size: UsiZeT,
57        source: *const u8,
58        source_size: UsiZeT,
59        output_buf: *mut u8,
60        output_size: *mut UsiZeT,
61        avail_output: UsiZeT,
62        flags: c_int,
63    ) -> c_int;
64}
