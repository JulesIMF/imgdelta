7use num_enum::TryFromPrimitive;
8use serde::{de, Deserialize, Deserializer, Serialize, Serializer};

10/// One-byte code identifying the file-level patch algorithm stored in a [`PatchRef`].
11///
12/// Using a fixed-width code instead of a string saves 6–15 bytes per manifest
13/// entry.  [`AlgorithmCode::Extended`] (`0xFF`) signals that the numeric code
14/// space is exhausted and the string `algorithm_id` field in [`PatchRef`] must
15/// be consulted instead.  No built-in algorithm currently uses `Extended`.
16///
17/// Values `0x03..=0xFE` are reserved for future built-in algorithms.
18///
19/// [`PatchRef`]: crate::PatchRef
20#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, TryFromPrimitive)]
21#[repr(u8)]
22pub enum AlgorithmCode {
23    /// No compression — target bytes stored verbatim (`"passthrough"`).
24    Passthrough = 0x00,
25    /// VCDIFF encoding via xdelta3 (`"xdelta3"`).
26    Xdelta3 = 0x01,
27    /// Myers line-level diff via the `diffy` crate (`"text-diff"`).
28    TextDiff = 0x02,
29    // 0x03..=0xFE reserved for future built-in algorithms
30    /// Code space exhausted — consult the string `algorithm_id` in [`PatchRef`].
31    Extended = 0xFF,
32}
33
34impl AlgorithmCode {
35    /// Return the raw byte value of this code.
36    #[inline]
37    pub fn as_u8(self) -> u8 {
38        self as u8
39    }
40}
41
42impl Serialize for AlgorithmCode {
43    fn serialize<S: Serializer>(&self, s: S) -> Result<S::Ok, S::Error> {
44        s.serialize_u8(self.as_u8())
45    }
46}
47
48impl<'de> Deserialize<'de> for AlgorithmCode {
49    fn deserialize<D: Deserializer<'de>>(d: D) -> Result<Self, D::Error> {
50        let v = u8::deserialize(d)?;
51        Self::try_from(v)
52            .map_err(|e| de::Error::custom(format!("unknown algorithm code: {:#04x}", e.number)))
53    }
54}

58/// A snapshot of a single file: its identity metadata and raw bytes.
59///
60/// [`PatchEncoder::encode`] receives **two** `FileSnapshot`s — one for the
61/// base version and one for the target version — so both sides are described
62/// symmetrically with the same struct.
63///
64/// Routing encoders (e.g. [`RouterEncoder`]) use `path`, `size`, and `header`
65/// from the **target** snapshot to select the right algorithm.  Concrete
66/// encoders (e.g. [`Xdelta3Encoder`]) use only the `bytes` field from each
67/// snapshot and ignore the routing fields.
68///
69/// [`PatchEncoder::encode`]: crate::PatchEncoder::encode
70/// [`RouterEncoder`]: crate::RouterEncoder
71/// [`Xdelta3Encoder`]: crate::Xdelta3Encoder
72pub struct FileSnapshot<'a> {
73    /// File path relative to the image root, forward-slash separated.
74    ///
75    /// Used by glob-based routing rules.
76    pub path: &'a str,
77    /// Uncompressed file size in bytes.
78    ///
79    /// Used by size-based routing rules.
80    pub size: u64,
81    /// First bytes of the file content (up to 16 bytes), for magic/ELF detection.
82    ///
83    /// Used by magic-byte and ELF routing rules.
84    pub header: &'a [u8],
85    /// Raw file content.
86    pub bytes: &'a [u8],
87}

89// ── FilePatch ─────────────────────────────────────────────────────────────────
90
91/// The result of [`PatchEncoder::encode`]: patch bytes together with the
92/// algorithm that produced them.
93///
94/// Bundling the algorithm with the bytes means callers—including
95/// [`RouterEncoder`]—can build a correct [`PatchRef`] without a second call
96/// to the encoder that was actually selected.
97///
98/// [`PatchEncoder::encode`]: crate::PatchEncoder::encode
99/// [`RouterEncoder`]: crate::RouterEncoder
100/// [`PatchRef`]: crate::PatchRef
101#[derive(Debug)]
102pub struct FilePatch {
103    /// Raw patch bytes to archive and later pass to [`PatchEncoder::decode`].
104    ///
105    /// [`PatchEncoder::decode`]: crate::PatchEncoder::decode
106    pub bytes: Vec<u8>,
107    /// Compact one-byte algorithm code.
108    pub code: AlgorithmCode,
109    /// Human-readable algorithm id — `Some` only when
110    /// `code == AlgorithmCode::Extended`.
111    pub algorithm_id: Option<String>,
112}
113
114impl FilePatch {
115    /// Construct a `FilePatch` for a built-in algorithm (code ≠ `Extended`).
116    ///
117    /// # Panics (debug only)
118    ///
119    /// Debug-asserts that `code != AlgorithmCode::Extended`.
120    /// Use [`FilePatch::extended`] for extended algorithms.
121    pub fn new(bytes: Vec<u8>, code: AlgorithmCode) -> Self {
122        debug_assert_ne!(
123            code,
124            AlgorithmCode::Extended,
125            "use FilePatch::extended for extended algorithms"
126        );
127        Self {
128            bytes,
129            code,
130            algorithm_id: None,
131        }
132    }
133
134    /// Construct a `FilePatch` for an extended (non-built-in) algorithm.
135    pub fn extended(bytes: Vec<u8>, algorithm_id: impl Into<String>) -> Self {
136        Self {
137            bytes,
138            code: AlgorithmCode::Extended,
139            algorithm_id: Some(algorithm_id.into()),
140        }
141    }
142}
