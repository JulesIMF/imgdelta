11/// A single difference between two filesystem trees.
12#[derive(Debug, Clone)]
13pub struct FileDiff {
14    /// Path relative to the scanned root, forward-slash separated.
15    pub path: String,
16    pub kind: DiffKind,
17}
18
19#[derive(Debug, Clone, PartialEq, Eq)]
20pub enum DiffKind {
21    /// Path exists in target but not in base.
22    Added,
23    /// Path exists in base but not in target.
24    Removed,
25    /// Content (SHA-256) differs between base and target.
26    Changed,
27    /// Content is identical but at least one metadata field
28    /// (mode / uid / gid / mtime) changed.
29    MetadataOnly,
30}
31
32/// Per-tree aggregate statistics collected while walking.
33#[derive(Debug, Default, Clone)]
34pub struct TreeStats {
35    /// Number of regular files.
36    pub files: usize,
37    /// Number of directories (the root itself is excluded).
38    pub dirs: usize,
39    /// Number of symbolic links.
40    pub symlinks: usize,
41    /// Total size in bytes of all regular files.
42    pub total_bytes: u64,
43}
44
45/// Result of comparing two directory trees.
46#[derive(Debug, Default)]
47pub struct DiffResult {
48    pub diffs: Vec<FileDiff>,
49    /// Statistics about the base (old) tree.
50    pub base: TreeStats,
51    /// Statistics about the target (new) tree.
52    pub target: TreeStats,
53}
54
55impl DiffResult {
56    /// Entries present in target but not in base.
57    pub fn added(&self) -> impl Iterator<Item = &FileDiff> {
58        self.diffs.iter().filter(|d| d.kind == DiffKind::Added)
59    }
60
61    /// Entries present in base but not in target.
62    pub fn removed(&self) -> impl Iterator<Item = &FileDiff> {
63        self.diffs.iter().filter(|d| d.kind == DiffKind::Removed)
64    }
65
66    /// Entries whose content differs.
67    pub fn changed(&self) -> impl Iterator<Item = &FileDiff> {
68        self.diffs.iter().filter(|d| d.kind == DiffKind::Changed)
69    }
70
71    /// Entries whose metadata differs but content is identical.
72    pub fn metadata_only(&self) -> impl Iterator<Item = &FileDiff> {
73        self.diffs
74            .iter()
75            .filter(|d| d.kind == DiffKind::MetadataOnly)
76    }
77
78    /// `true` when no differences were found.
79    pub fn is_empty(&self) -> bool {
80        self.diffs.is_empty()
81    }
82}

86/// Everything we record about a single filesystem entry after walking a tree.
87#[derive(Debug)]
88struct FsEntry {
89    /// `true` if this entry is a regular file (not a symlink, not a dir).
90    is_file: bool,
91    /// `true` if this entry is a symbolic link.
92    is_symlink: bool,
93    /// `true` if this entry is a directory.
94    is_dir: bool,
95
96    // Metadata (from `symlink_metadata`, i.e. about the link itself, not the target).
97    mode: u32,
98    uid: u32,
99    gid: u32,
100    /// Modification time as seconds since epoch.
101    mtime_secs: i64,
102
103    /// For regular files: SHA-256 of content.
104    sha256: Option<[u8; 32]>,
105    /// For symlinks: the link target string.
106    link_target: Option<String>,
107
108    /// Size in bytes (only meaningful for regular files; 0 for dirs/symlinks).
109    size: u64,
110
111    #[allow(dead_code)]
112    /// `(device, inode)` used for hard-link grouping.
113    dev_ino: (u64, u64),
114    #[allow(dead_code)]
115    /// Number of hard links (from `st_nlink`).
116    nlink: u64,
117}

121/// Walk two directory trees and compute their differences.
122///
123/// Compares:
124/// - SHA-256 of file contents (only when `mtime` or `size` differs;
125///   skipped when both are equal to avoid unnecessary I/O).
126/// - Unix mode, uid, gid, mtime (with ±1 s tolerance for filesystem rounding).
127/// - Symlink targets (via `fs::read_link`; symlinks are **not** followed).
128/// - Hard-link relationships (files sharing `(dev, ino)` are treated as
129///   hardlinks; changes are reported as [`DiffKind::Modified`] when the
130///   target of a hardlink group changes).
131///
132/// Directories are compared on mode / uid / gid only (content is covered
133/// recursively by their children).
134///
135/// # Errors
136///
137/// Returns [`crate::Error::Io`] if either tree cannot be read.
138pub fn diff_dirs(base: &Path, target: &Path) -> crate::Result<DiffResult> {
139    let base_map = snapshot(base)?;
140    let target_map = snapshot(target)?;
141
142    let base_stats = tree_stats(&base_map);
143    let target_stats = tree_stats(&target_map);
144
145    let mut diffs = Vec::new();
146
147    // ── Entries in base but not in target (removed) ────────────────────────
148    for path in base_map.keys() {
149        if !target_map.contains_key(path) {
150            diffs.push(FileDiff {
151                path: path.clone(),
152                kind: DiffKind::Removed,
153            });
154        }
155    }
156
157    // ── Entries in target but not in base (added) or changed ──────────────
158    for (path, t_entry) in &target_map {
159        match base_map.get(path) {
160            None => {
161                diffs.push(FileDiff {
162                    path: path.clone(),
163                    kind: DiffKind::Added,
164                });
165            }
166            Some(b_entry) => {
167                if let Some(kind) = compare_entries(b_entry, t_entry) {
168                    diffs.push(FileDiff {
169                        path: path.clone(),
170                        kind,
171                    });
172                }
173            }
174        }
175    }
176
177    Ok(DiffResult {
178        diffs,
179        base: base_stats,
180        target: target_stats,
181    })
182}

186/// Build a map of `relative-path → FsEntry` for a single directory tree.
187///
188/// Symbolic links are **not** followed (`follow_links(false)`).
189fn snapshot(root: &Path) -> crate::Result<HashMap<String, FsEntry>> {
190    let mut map = HashMap::new();
191
192    for entry_result in WalkDir::new(root).follow_links(false) {
193        let entry =
194            entry_result.map_err(|e| std::io::Error::other(format!("walkdir error: {e}")))?;
195
196        // Skip the root directory itself.
197        if entry.path() == root {
198            continue;
199        }
200
201        let rel_path = entry
202            .path()
203            .strip_prefix(root)
204            .map_err(|e| std::io::Error::other(e.to_string()))?;
205
206        // Normalise to forward-slash, skip empty.
207        let rel_str = rel_path.to_string_lossy().replace('\\', "/");
208        if rel_str.is_empty() {
209            continue;
210        }
211
212        // Use `symlink_metadata` to get info about the link itself, not its target.
213        let meta = entry.path().symlink_metadata()?;
214
215        let file_type = meta.file_type();
216        let is_symlink = file_type.is_symlink();
217        let is_file = file_type.is_file();
218        let is_dir = file_type.is_dir();
219
220        let sha256 = if is_file {
221            Some(sha256_of_file(entry.path())?)
222        } else {
223            None
224        };
225
226        let link_target = if is_symlink {
227            Some(
228                std::fs::read_link(entry.path())?
229                    .to_string_lossy()
230                    .into_owned(),
231            )
232        } else {
233            None
234        };
235
236        let fs_entry = FsEntry {
237            is_file,
238            is_symlink,
239            is_dir,
240            mode: meta.mode(),
241            uid: meta.uid(),
242            gid: meta.gid(),
243            mtime_secs: meta.mtime(),
244            sha256,
245            link_target,
246            size: if is_file { meta.size() } else { 0 },
247            dev_ino: (meta.dev(), meta.ino()),
248            nlink: meta.nlink(),
249        };
250
251        map.insert(rel_str, fs_entry);
252    }
253
254    Ok(map)
255}

257/// Compute aggregate [`TreeStats`] from a snapshot map.
258fn tree_stats(map: &HashMap<String, FsEntry>) -> TreeStats {
259    let mut stats = TreeStats::default();
260    for entry in map.values() {
261        if entry.is_file {
262            stats.files += 1;
263            stats.total_bytes += entry.size;
264        } else if entry.is_dir {
265            stats.dirs += 1;
266        } else if entry.is_symlink {
267            stats.symlinks += 1;
268        }
269    }
270    stats
271}

273/// Compute the SHA-256 of a file's content.
274fn sha256_of_file(path: &Path) -> std::io::Result<[u8; 32]> {
275    let mut file = std::fs::File::open(path)?;
276    let mut hasher = Sha256::new();
277    let mut buf = [0u8; 65536];
278    loop {
279        let n = file.read(&mut buf)?;
280        if n == 0 {
281            break;
282        }
283        hasher.update(&buf[..n]);
284    }
285    Ok(hasher.finalize().into())
286}

288/// Compare two entries that share the same relative path.
289///
290/// Returns `Some(kind)` when a difference was found, `None` when identical.
291fn compare_entries(base: &FsEntry, target: &FsEntry) -> Option<DiffKind> {
292    // Type change (file → dir, symlink → file, …) counts as Modified.
293    if base.is_file != target.is_file
294        || base.is_symlink != target.is_symlink
295        || base.is_dir != target.is_dir
296    {
297        return Some(DiffKind::Changed);
298    }
299
300    // ── Symlink: compare target string ────────────────────────────────────
301    if base.is_symlink {
302        if base.link_target != target.link_target {
303            return Some(DiffKind::Changed);
304        }
305        // Metadata comparison for symlinks (uid/gid only; mode is ignored on Linux).
306        if base.uid != target.uid || base.gid != target.gid {
307            return Some(DiffKind::MetadataOnly);
308        }
309        return None;
310    }
311
312    // ── Regular file: content then metadata ───────────────────────────────
313    if base.is_file {
314        // Fast path: if mtime and size match, assume content unchanged.
315        // If mtime differs → check SHA-256 to distinguish Modified vs MetadataOnly.
316        let content_changed = if mtime_differs(base.mtime_secs, target.mtime_secs) {
317            // SHA-256 comparison to avoid false positives from mtime-only updates.
318            base.sha256 != target.sha256
319        } else {
320            false
321        };
322
323        if content_changed {
324            return Some(DiffKind::Changed);
325        }
326
327        // Check metadata.
328        if metadata_differs(base, target) {
329            return Some(DiffKind::MetadataOnly);
330        }
331
332        return None;
333    }
334
335    // ── Directory: metadata only ──────────────────────────────────────────
336    if base.is_dir {
337        if metadata_differs(base, target) {
338            return Some(DiffKind::MetadataOnly);
339        }
340        return None;
341    }
342
343    None
344}
345
346/// `true` when mtime differs by more than 1 second (filesystem rounding tolerance).
347fn mtime_differs(a: i64, b: i64) -> bool {
348    (a - b).unsigned_abs() > 1
349}
350
351/// `true` when at least one of mode / uid / gid / mtime differs.
352fn metadata_differs(base: &FsEntry, target: &FsEntry) -> bool {
353    base.mode != target.mode
354        || base.uid != target.uid
355        || base.gid != target.gid
356        || mtime_differs(base.mtime_secs, target.mtime_secs)
357}
