7//! Three-stage stateless decompress pipeline for one `Fs` partition.
8//!
9//! Stages:
10//! 1. [`stages::ExtractArchive`] — decompress/index the patches tar archive
11//! 2. [`stages::CopyUnchanged`] — copy unchanged base files to output
12//! 3. [`stages::ApplyRecords`] — download blobs + apply all manifest records
13//!
14//! The public entry point [`decompress_fs_partition`] has the same signature as
15//! the old `decompress_pipeline::decompress_fs_partition` and is a drop-in
16//! replacement.
17
18pub mod context;
19pub mod draft;
20pub mod pipeline;
21pub mod stage;
22pub mod stages;

48/// Reconstruct an Fs partition into `output_root` from `base_root` + manifest records.
49///
50/// Drop-in replacement for `decompress_pipeline::decompress_fs_partition`.
51#[allow(clippy::too_many_arguments)]
52pub async fn decompress_fs_partition(
53    base_root: &Path,
54    output_root: &Path,
55    records: &[Record],
56    archive_bytes: &[u8],
57    patches_compressed: bool,
58    storage: Arc<dyn Storage>,
59    router: Arc<RouterEncoder>,
60    workers: usize,
61) -> Result<PartitionDecompressStats> {
62    let ctx = DecompressContext {
63        storage,
64        router,
65        workers,
66        base_root: Arc::from(base_root),
67        output_root: Arc::from(output_root),
68        records: Arc::from(records),
69        archive_bytes: Arc::from(archive_bytes),
70        patches_compressed,
71    };
72
73    let pipeline = DecompressPipeline::default_fs();
74    let draft = pipeline.run(&ctx, DecompressDraft::default()).await?;
75
76    Ok(draft.stats)
77}
