27/// Handles decompression for a single partition, regardless of type.
28///
29/// Implementations exist for:
30/// - [`FsPartitionDecompressor`]        — runs the 3-stage file-system decompress pipeline.
31/// - [`BiosBootDecompressor`]           — downloads blob and calls `write_raw()` on the handle.
32/// - [`RawPartitionDecompressor`]       — downloads blob and calls `write_raw()` on the handle.
33/// - [`MbrDecompressor`]                — downloads blob and calls `write_raw()` on the handle.
34#[async_trait]
35pub trait PartitionDecompressor: Send + Sync {
36    /// Decompress one partition manifest into `output_ph`.
37    ///
38    /// All shared inputs (storage, router, workers, patch map, base FS handles)
39    /// are contained in `ctx`.  `pm` is the partition manifest to decompress;
40    /// `output_ph` must be a **writable** [`PartitionHandle`]:
41    /// - `PartitionHandle::Fs`       — its `mount_fn` must return an RW mount.
42    /// - `PartitionHandle::BiosBoot` — must have a `write_fn` set.
43    /// - `PartitionHandle::Raw`      — must have a `write_fn` set.
44    /// - `PartitionHandle::Mbr`      — must have a `write_fn` set.
45    async fn decompress(
46        &self,
47        ctx: &DecompressContext,
48        pm: &PartitionManifest,
49        output_ph: &PartitionHandle,
50    ) -> Result<PartitionDecompressStats>;
51}
