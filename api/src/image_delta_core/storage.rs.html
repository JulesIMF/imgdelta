10/// A blob candidate returned by storage when searching for a suitable delta base.
11///
12/// Returned by [`Storage::find_blob_candidates`] and consumed by the path-matcher
13/// to decide which base blob to use for a delta.
14#[derive(Debug, Clone)]
15pub struct BlobCandidate {
16    /// UUID assigned when the blob was uploaded; used to call
17    /// [`Storage::download_blob`].
18    pub uuid: Uuid,
19    /// SHA-256 hex digest of the blob bytes.  Used to verify integrity after
20    /// download.
21    pub sha256: String,
22    /// Relative path this blob was recorded under via
23    /// [`Storage::record_blob_origin`].  Used by the path-matcher for scoring.
24    pub original_path: String,
25}

27/// Lightweight metadata about an image known to storage.
28#[derive(Debug, Clone)]
29pub struct ImageMeta {
30    /// Provider-assigned identifier for this image.
31    pub image_id: String,
32    /// Identifier of the base image used as delta source, or `None` for root images.
33    pub base_image_id: Option<String>,
34    /// Image container format (`"directory"`, `"qcow2"`, …).
35    pub format: String,
36    /// Current lifecycle status string, e.g. `"pending"`, `"compressing"`,
37    /// `"compressed"`, `"failed"`.
38    pub status: String,
39}
40
41/// Lifecycle state of an image in storage.
42#[derive(Debug, Clone)]
43pub enum ImageStatus {
44    /// Compression job created but not yet started.
45    Pending,
46    /// Compression is currently in progress.
47    Compressing,
48    /// Compression finished successfully; patches are available.
49    Compressed,
50    /// Compression failed; the inner string contains a human-readable reason.
51    Failed(String),
52}

54/// Persistent storage backend abstraction.
55///
56/// S3 + PostgreSQL is the production implementation (in `image-delta-cli`).
57/// [`FakeStorage`] (in-memory) is used for L1 unit/integration tests.
58///
59/// # Contract for implementors
60///
61/// - All methods must be safe to call concurrently from multiple threads.
62/// - `upload_blob` must be idempotent: uploading the same bytes twice (same SHA-256)
63///   must return the same UUID and not store a duplicate.
64///
65/// # Note
66///
67/// `save_stats` from the original design is intentionally omitted here to avoid a
68/// circular dependency with `compressor::CompressionStats`.  It will be added in
69/// Phase 5 via a separate `StoredStats` type.
70#[async_trait]
71pub trait Storage: Send + Sync {
72    // ── Blob CAS ─────────────────────────────────────────────────────────────
73
74    /// Check whether a blob with this SHA-256 digest already exists.
75    ///
76    /// Returns `Some(uuid)` if it exists, `None` otherwise.
77    async fn blob_exists(&self, sha256: &str) -> crate::Result<Option<Uuid>>;
78
79    /// Upload raw bytes, keyed by their SHA-256 hex digest.
80    ///
81    /// Must be idempotent: if a blob with `sha256` already exists, returns the
82    /// existing UUID without re-uploading.
83    async fn upload_blob(&self, sha256: &str, data: &[u8]) -> crate::Result<Uuid>;
84
85    /// Download raw bytes for a known blob UUID.
86    async fn download_blob(&self, blob_id: Uuid) -> crate::Result<Vec<u8>>;
87
88    // ── Image data ────────────────────────────────────────────────────────────
89
90    /// Upload a serialised manifest for `image_id`.
91    async fn upload_manifest(&self, image_id: &str, manifest_bytes: &[u8]) -> crate::Result<()>;
92
93    /// Download the serialised manifest for `image_id`.
94    async fn download_manifest(&self, image_id: &str) -> crate::Result<Vec<u8>>;
95
96    /// Upload the patches tar (or tar.gz) archive for `image_id`.
97    ///
98    /// `compressed` — `true` if the bytes are gzip-compressed (tar.gz).
99    async fn upload_patches(
100        &self,
101        image_id: &str,
102        data: &[u8],
103        compressed: bool,
104    ) -> crate::Result<()>;
105
106    /// Download the patches tar (or tar.gz) archive for `image_id`.
107    async fn download_patches(&self, image_id: &str) -> crate::Result<Vec<u8>>;
108
109    // ── DB ────────────────────────────────────────────────────────────────────
110
111    /// Persist metadata for a newly created or updated image.
112    async fn register_image(&self, meta: &ImageMeta) -> crate::Result<()>;
113
114    /// Look up metadata for an image by ID, returning `None` if not found.
115    async fn get_image(&self, image_id: &str) -> crate::Result<Option<ImageMeta>>;
116
117    /// Update the lifecycle status of an image.
118    async fn update_status(&self, image_id: &str, status: ImageStatus) -> crate::Result<()>;
119
120    /// Return metadata for all known images.
121    async fn list_images(&self) -> crate::Result<Vec<ImageMeta>>;
122
123    // ── BlobPatch ─────────────────────────────────────────────────────────────
124
125    /// Return blobs from `base_image_id` that are candidates for delta encoding
126    /// against files in the new image.
127    ///
128    /// When `partition_number` is `Some(n)`, only blobs recorded for partition
129    /// `n` are returned (i.e. same-partition candidates for multi-partition
130    /// images).  Pass `None` to return blobs from all partitions.
131    async fn find_blob_candidates(
132        &self,
133        base_image_id: &str,
134        partition_number: Option<i32>,
135    ) -> crate::Result<Vec<BlobCandidate>>;
136
137    /// Record that `blob_uuid` originated from `file_path` in `orig_image_id`.
138    ///
139    /// `base_image_id` is the delta-source image that was used during the
140    /// compress run that produced this blob (i.e. `options.base_image_id`).
141    /// It is `None` for root (no-base) compress runs.
142    ///
143    /// `partition_number` is the 1-based partition number in the source image.
144    /// `None` for single-partition (`DirectoryImage`) images.
145    ///
146    /// Called by [`DefaultCompressor`] after each `upload_blob` so that future
147    /// compress operations can use this blob as a delta base via
148    /// [`find_blob_candidates`].
149    ///
150    /// [`DefaultCompressor`]: crate::DefaultCompressor
151    /// [`find_blob_candidates`]: Storage::find_blob_candidates
152    async fn record_blob_origin(
153        &self,
154        blob_uuid: Uuid,
155        orig_image_id: &str,
156        base_image_id: Option<&str>,
157        partition_number: Option<i32>,
158        file_path: &str,
159    ) -> crate::Result<()>;
160
161    // ── Delete ────────────────────────────────────────────────────────────────
162
163    /// Delete the manifest object for `image_id`.
164    async fn delete_manifest(&self, image_id: &str) -> crate::Result<()>;
165
166    /// Delete the patches archive for `image_id`.
167    async fn delete_patches(&self, image_id: &str) -> crate::Result<()>;
168
169    /// Delete a single blob by UUID.  Called only when ref-counting confirms the
170    /// blob is not referenced by any other image.
171    async fn delete_blob(&self, blob_id: Uuid) -> crate::Result<()>;
172
173    /// Remove all `blob_origins` rows where `orig_image_id = image_id`.
174    async fn delete_blob_origins(&self, image_id: &str) -> crate::Result<()>;
175
176    /// Remove the image metadata record (images table / meta.json).
177    async fn delete_image_meta(&self, image_id: &str) -> crate::Result<()>;
178}
