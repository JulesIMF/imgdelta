29/// Persistent storage backend abstraction.
30///
31/// S3 + PostgreSQL is the production implementation (in `image-delta-cli`).
32/// [`FakeStorage`] (in-memory) is used for L1 unit/integration tests.
33///
34/// # Contract for implementors
35///
36/// - All methods must be safe to call concurrently from multiple threads.
37/// - `upload_blob` must be idempotent: uploading the same bytes twice is fine.
38pub trait Storage: Send + Sync {
39    /// Upload raw bytes and return the assigned UUID.
40    fn upload_blob(&self, data: &[u8]) -> crate::Result<Uuid>;
41
42    /// Download raw bytes for a known blob UUID.
43    fn download_blob(&self, blob_id: Uuid) -> crate::Result<Vec<u8>>;
44
45    /// Upload a serialised manifest for `image_id`.
46    fn upload_manifest(&self, image_id: &str, manifest_bytes: &[u8]) -> crate::Result<()>;
47
48    /// Download the serialised manifest for `image_id`.
49    fn download_manifest(&self, image_id: &str) -> crate::Result<Vec<u8>>;
50
51    /// Return blobs from `base_image_id` that are candidates for delta encoding
52    /// against files in the new image.
53    fn find_blob_candidates(&self, base_image_id: &str) -> crate::Result<Vec<BlobCandidate>>;
54
55    fn save_image_meta(&self, meta: &ImageMeta) -> crate::Result<()>;
56
57    fn get_image_meta(&self, image_id: &str) -> crate::Result<Option<ImageMeta>>;
58
59    fn set_image_status(&self, image_id: &str, status: ImageStatus) -> crate::Result<()>;
60
61    fn list_images(&self) -> crate::Result<Vec<ImageMeta>>;
62
63    /// Upload the patches tar (or tar.gz) archive for `image_id`.
64    fn upload_patches(&self, image_id: &str, data: &[u8]) -> crate::Result<()>;
65
66    /// Download the patches tar (or tar.gz) archive for `image_id`.
67    fn download_patches(&self, image_id: &str) -> crate::Result<Vec<u8>>;
68}
