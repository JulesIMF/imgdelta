11/// Low-level encoding/decoding algorithm.
12///
13/// This trait is the raw worker — it receives only bytes, applies the
14/// algorithm, and returns bytes.  It knows nothing about file metadata,
15/// routing, or manifest bookkeeping.
16///
17/// # Implementing a new algorithm
18///
19/// 1. Create a new `pub(crate)` struct inside the relevant `encoders/` module.
20/// 2. Implement this trait for it.
21/// 3. Wrap it with a public `PatchEncoder` implementation that owns an instance
22///    (or creates one per call) and maps the algorithm code.
23///
24/// # Instantiation cost
25///
26/// Implementors should make `new()` as cheap as possible: a [`PatchEncoder`]
27/// may create a fresh algorithm instance for **every file** it encodes.
28/// Zero-size types (`struct Xdelta3Algorithm;`) are ideal.
29pub(crate) trait PatchAlgorithm {
30    /// Encode `target` relative to `source`, returning raw patch bytes.
31    fn encode_raw(&self, source: &[u8], target: &[u8]) -> Result<Vec<u8>>;
32    /// Reconstruct `target` from `source` and previously produced patch bytes.
33    fn decode_raw(&self, source: &[u8], patch: &[u8]) -> Result<Vec<u8>>;
34}

36// ── PatchEncoder ──────────────────────────────────────────────────────────────

38/// Trait for computing and applying file-level binary patches.
39///
40/// In this project, "patch" refers to a file-level diff (what this trait
41/// produces), while "delta" refers to the full image-level diff produced by
42/// [`DefaultCompressor`].
43///
44/// An implementor receives two [`FileSnapshot`]s — one for the base version
45/// and one for the target version.  Both are described with the same struct,
46/// each carrying file-level metadata (`path`, `size`, `header`) and raw bytes.
47/// Concrete encoders (e.g. [`Xdelta3Encoder`]) use only the `bytes` fields.
48/// Routing encoders (e.g. [`RouterEncoder`]) use the target metadata to
49/// dispatch to the right concrete encoder before delegating.
50///
51/// `decode` receives the source bytes and a [`FilePatch`] (which carries both
52/// the raw patch bytes and the algorithm code), so routing encoders can select
53/// the correct decoder without a separate lookup call from the caller.
54///
55/// # Contract for implementors
56///
57/// - `decode(base.bytes, encode(base, target)?) == target.bytes` for all inputs.
58/// - Implementations must be `Send + Sync` (used from multiple worker threads).
59/// - `algorithm_code()` must be stable across versions — it is the primary key
60///   stored in the manifest for decoder selection during decompression.
61///   Return `None` only for routing/delegating encoders that have no fixed algorithm.
62/// - `algorithm_id()` must also be stable — it is the fallback key when
63///   `algorithm_code() == Some(AlgorithmCode::Extended)`.
64///
65/// [`DefaultCompressor`]: crate::DefaultCompressor
66/// [`Xdelta3Encoder`]: crate::Xdelta3Encoder
67/// [`RouterEncoder`]: crate::RouterEncoder
68pub trait PatchEncoder: Send + Sync {
69    /// Compute a binary patch from `base` to `target`.
70    ///
71    /// Both `base` and `target` carry the same metadata fields; routing
72    /// encoders use `target.path`, `target.size`, and `target.header` to
73    /// select the right algorithm.  Concrete encoders use `base.bytes` and
74    /// `target.bytes` only.
75    ///
76    /// Returns a [`FilePatch`] that bundles the patch bytes with the algorithm
77    /// that produced them, so callers can construct a correct [`PatchRef`]
78    /// without a second call to the encoder.
79    ///
80    /// # Errors
81    ///
82    /// Returns [`Error::Encode`] if encoding fails.
83    ///
84    /// [`PatchRef`]: crate::PatchRef
85    fn encode(&self, base: &FileSnapshot<'_>, target: &FileSnapshot<'_>) -> Result<FilePatch>;
86
87    /// Reconstruct the target from `source` bytes and a previously computed [`FilePatch`].
88    ///
89    /// The [`FilePatch`] carries both the raw patch bytes (`patch.bytes`) and
90    /// the algorithm code (`patch.code`), which routing encoders use to select
91    /// the correct concrete decoder without a separate lookup call.
92    ///
93    /// # Errors
94    ///
95    /// Returns [`Error::Decode`] if the patch is invalid or corrupt.
96    fn decode(&self, source: &[u8], patch: &FilePatch) -> Result<Vec<u8>>;
97
98    /// Compact one-byte algorithm code — the primary lookup key during decompression.
99    ///
100    /// Returns `None` for routing/delegating encoders (e.g. [`RouterEncoder`])
101    /// that do not have a single fixed algorithm.
102    ///
103    /// Returns `Some(AlgorithmCode::Extended)` for non-built-in algorithms, in
104    /// which case [`algorithm_id`] is used as the fallback.
105    ///
106    /// [`algorithm_id`]: PatchEncoder::algorithm_id
107    /// [`RouterEncoder`]: crate::RouterEncoder
108    fn algorithm_code(&self) -> Option<AlgorithmCode>;
109
110    /// Stable string identifier.
111    ///
112    /// Stored in [`PatchRef`] only when
113    /// `algorithm_code() == Some(AlgorithmCode::Extended)`.  For built-in
114    /// algorithms this string is informational only.
115    ///
116    /// [`PatchRef`]: crate::PatchRef
117    fn algorithm_id(&self) -> &'static str;
118}
