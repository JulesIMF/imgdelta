25/// Low-level encoding/decoding algorithm.
26///
27/// This trait is the raw worker — it receives only bytes, applies the
28/// algorithm, and returns bytes.  It knows nothing about file metadata,
29/// routing, or manifest bookkeeping.
30///
31/// # Implementing a new algorithm
32///
33/// 1. Create a new `pub(crate)` struct inside the relevant `encoders/` module.
34/// 2. Implement this trait for it.
35/// 3. Wrap it with a public `PatchEncoder` implementation that owns an instance
36///    (or creates one per call) and maps the algorithm code.
37///
38/// # Instantiation cost
39///
40/// Implementors should make `new()` as cheap as possible: a [`PatchEncoder`]
41/// may create a fresh algorithm instance for **every file** it encodes.
42/// Zero-size types (`struct Xdelta3Algorithm;`) are ideal.
43pub(crate) trait PatchAlgorithm {
44    /// Encode `target` relative to `source`, returning raw patch bytes.
45    fn encode_raw(&self, source: &[u8], target: &[u8]) -> Result<Vec<u8>>;
46    /// Reconstruct `target` from `source` and previously produced patch bytes.
47    fn decode_raw(&self, source: &[u8], patch: &[u8]) -> Result<Vec<u8>>;
48}
49
50/// One-byte code identifying the file-level patch algorithm stored in a [`PatchRef`].
51///
52/// Using a fixed-width code instead of a string saves 6–15 bytes per manifest
53/// entry.  [`AlgorithmCode::Extended`] (`0xFF`) signals that the numeric code
54/// space is exhausted and the string `algorithm_id` field in [`PatchRef`] must
55/// be consulted instead.  No built-in algorithm currently uses `Extended`.
56///
57/// Values `0x03..=0xFE` are reserved for future built-in algorithms.
58///
59/// [`PatchRef`]: crate::PatchRef
60#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, TryFromPrimitive)]
61#[repr(u8)]
62pub enum AlgorithmCode {
63    /// No compression — target bytes stored verbatim (`"passthrough"`).
64    Passthrough = 0x00,
65    /// VCDIFF encoding via xdelta3 (`"xdelta3"`).
66    Xdelta3 = 0x01,
67    /// Myers line-level diff via the `diffy` crate (`"text-diff"`).
68    TextDiff = 0x02,
69    // 0x03..=0xFE reserved for future built-in algorithms
70    /// Code space exhausted — consult the string `algorithm_id` in [`PatchRef`].
71    Extended = 0xFF,
72}
73
74impl AlgorithmCode {
75    /// Return the raw byte value of this code.
76    #[inline]
77    pub fn as_u8(self) -> u8 {
78        self as u8
79    }
80}
81
82impl Serialize for AlgorithmCode {
83    fn serialize<S: Serializer>(&self, s: S) -> std::result::Result<S::Ok, S::Error> {
84        s.serialize_u8(self.as_u8())
85    }
86}
87
88impl<'de> Deserialize<'de> for AlgorithmCode {
89    fn deserialize<D: Deserializer<'de>>(d: D) -> std::result::Result<Self, D::Error> {
90        let v = u8::deserialize(d)?;
91        Self::try_from(v)
92            .map_err(|e| de::Error::custom(format!("unknown algorithm code: {:#04x}", e.number)))
93    }
94}

98/// A snapshot of a single file: its identity metadata and raw bytes.
99///
100/// [`PatchEncoder::encode`] receives **two** `FileSnapshot`s — one for the
101/// base version and one for the target version — so both sides are described
102/// symmetrically with the same struct.
103///
104/// Routing encoders (e.g. [`RouterEncoder`]) use `path`, `size`, and `header`
105/// from the **target** snapshot to select the right algorithm.  Concrete
106/// encoders (e.g. [`Xdelta3Encoder`]) use only the `bytes` field from each
107/// snapshot and ignore the routing fields.
108///
109/// [`PatchEncoder::encode`]: crate::PatchEncoder::encode
110/// [`RouterEncoder`]: crate::RouterEncoder
111/// [`Xdelta3Encoder`]: crate::Xdelta3Encoder
112pub struct FileSnapshot<'a> {
113    /// File path relative to the image root, forward-slash separated.
114    ///
115    /// Used by glob-based routing rules.
116    pub path: &'a str,
117    /// Uncompressed file size in bytes.
118    ///
119    /// Used by size-based routing rules.
120    pub size: u64,
121    /// First bytes of the file content (up to 16 bytes), for magic/ELF detection.
122    ///
123    /// Used by magic-byte and ELF routing rules.
124    pub header: &'a [u8],
125    /// Raw file content.
126    pub bytes: &'a [u8],
127}

129// ── FilePatch ─────────────────────────────────────────────────────────────────
130
131/// The result of [`PatchEncoder::encode`]: patch bytes together with the
132/// algorithm that produced them.
133///
134/// Bundling the algorithm with the bytes means callers—including
135/// [`RouterEncoder`]—can build a correct [`PatchRef`] without a second call
136/// to the encoder that was actually selected.
137///
138/// [`PatchEncoder::encode`]: crate::PatchEncoder::encode
139/// [`RouterEncoder`]: crate::RouterEncoder
140/// [`PatchRef`]: crate::PatchRef
141#[derive(Debug)]
142pub struct FilePatch {
143    /// Raw patch bytes to archive and later pass to [`PatchEncoder::decode`].
144    ///
145    /// [`PatchEncoder::decode`]: crate::PatchEncoder::decode
146    pub bytes: Vec<u8>,
147    /// Compact one-byte algorithm code.
148    pub code: AlgorithmCode,
149    /// Human-readable algorithm id — `Some` only when
150    /// `code == AlgorithmCode::Extended`.
151    pub algorithm_id: Option<String>,
152}
153
154impl FilePatch {
155    /// Construct a `FilePatch` for a built-in algorithm (code ≠ `Extended`).
156    ///
157    /// # Panics (debug only)
158    ///
159    /// Debug-asserts that `code != AlgorithmCode::Extended`.
160    /// Use [`FilePatch::extended`] for extended algorithms.
161    pub fn new(bytes: Vec<u8>, code: AlgorithmCode) -> Self {
162        debug_assert_ne!(
163            code,
164            AlgorithmCode::Extended,
165            "use FilePatch::extended for extended algorithms"
166        );
167        Self {
168            bytes,
169            code,
170            algorithm_id: None,
171        }
172    }
173
174    /// Construct a `FilePatch` for an extended (non-built-in) algorithm.
175    pub fn extended(bytes: Vec<u8>, algorithm_id: impl Into<String>) -> Self {
176        Self {
177            bytes,
178            code: AlgorithmCode::Extended,
179            algorithm_id: Some(algorithm_id.into()),
180        }
181    }
182}

184// ── PatchEncoder ──────────────────────────────────────────────────────────────

186/// Trait for computing and applying file-level binary patches.
187///
188/// In this project, "patch" refers to a file-level diff (what this trait
189/// produces), while "delta" refers to the full image-level diff produced by
190/// [`DefaultCompressor`].
191///
192/// An implementor receives two [`FileSnapshot`]s — one for the base version
193/// and one for the target version.  Both are described with the same struct,
194/// each carrying file-level metadata (`path`, `size`, `header`) and raw bytes.
195/// Concrete encoders (e.g. [`Xdelta3Encoder`]) use only the `bytes` fields.
196/// Routing encoders (e.g. [`RouterEncoder`]) use the target metadata to
197/// dispatch to the right concrete encoder before delegating.
198///
199/// `decode` receives the source bytes and a [`FilePatch`] (which carries both
200/// the raw patch bytes and the algorithm code), so routing encoders can select
201/// the correct decoder without a separate lookup call from the caller.
202///
203/// # Contract for implementors
204///
205/// - `decode(base.bytes, encode(base, target)?) == target.bytes` for all inputs.
206/// - Implementations must be `Send + Sync` (used from multiple worker threads).
207/// - `algorithm_code()` must be stable across versions — it is the primary key
208///   stored in the manifest for decoder selection during decompression.
209///   Return `None` only for routing/delegating encoders that have no fixed algorithm.
210/// - `algorithm_id()` must also be stable — it is the fallback key when
211///   `algorithm_code() == Some(AlgorithmCode::Extended)`.
212///
213/// [`DefaultCompressor`]: crate::DefaultCompressor
214/// [`Xdelta3Encoder`]: crate::Xdelta3Encoder
215/// [`RouterEncoder`]: crate::RouterEncoder
216pub trait PatchEncoder: Send + Sync {
217    /// Compute a binary patch from `base` to `target`.
218    ///
219    /// Both `base` and `target` carry the same metadata fields; routing
220    /// encoders use `target.path`, `target.size`, and `target.header` to
221    /// select the right algorithm.  Concrete encoders use `base.bytes` and
222    /// `target.bytes` only.
223    ///
224    /// Returns a [`FilePatch`] that bundles the patch bytes with the algorithm
225    /// that produced them, so callers can construct a correct [`PatchRef`]
226    /// without a second call to the encoder.
227    ///
228    /// # Errors
229    ///
230    /// Returns [`Error::Encode`] if encoding fails.
231    ///
232    /// [`PatchRef`]: crate::PatchRef
233    fn encode(&self, base: &FileSnapshot<'_>, target: &FileSnapshot<'_>) -> Result<FilePatch>;
234
235    /// Reconstruct the target from `source` bytes and a previously computed [`FilePatch`].
236    ///
237    /// The [`FilePatch`] carries both the raw patch bytes (`patch.bytes`) and
238    /// the algorithm code (`patch.code`), which routing encoders use to select
239    /// the correct concrete decoder without a separate lookup call.
240    ///
241    /// # Errors
242    ///
243    /// Returns [`Error::Decode`] if the patch is invalid or corrupt.
244    fn decode(&self, source: &[u8], patch: &FilePatch) -> Result<Vec<u8>>;
245
246    /// Compact one-byte algorithm code — the primary lookup key during decompression.
247    ///
248    /// Returns `None` for routing/delegating encoders (e.g. [`RouterEncoder`])
249    /// that do not have a single fixed algorithm.
250    ///
251    /// Returns `Some(AlgorithmCode::Extended)` for non-built-in algorithms, in
252    /// which case [`algorithm_id`] is used as the fallback.
253    ///
254    /// [`algorithm_id`]: PatchEncoder::algorithm_id
255    /// [`RouterEncoder`]: crate::RouterEncoder
256    fn algorithm_code(&self) -> Option<AlgorithmCode>;
257
258    /// Stable string identifier.
259    ///
260    /// Stored in [`PatchRef`] only when
261    /// `algorithm_code() == Some(AlgorithmCode::Extended)`.  For built-in
262    /// algorithms this string is informational only.
263    ///
264    /// [`PatchRef`]: crate::PatchRef
265    fn algorithm_id(&self) -> &'static str;
266}
