15/// [`Image`] implementation for a plain directory.
16///
17/// "Mounting" is a no-op: the directory is used as-is.  No external tools
18/// required.  This is the format used for all L1 and L2 tests, and for
19/// providers whose pipeline already extracts images to directories.
20pub struct DirectoryImage;
21
22impl DirectoryImage {
23    pub fn new() -> Self {
24        Self
25    }
26}
27
28impl Default for DirectoryImage {
29    fn default() -> Self {
30        Self::new()
31    }
32}
33
34impl Image for DirectoryImage {
35    fn format_name(&self) -> &'static str {
36        "directory"
37    }
38
39    fn open(&self, path: &Path) -> Result<Box<dyn OpenImage>> {
40        if !path.is_dir() {
41            return Err(crate::Error::Format(format!(
42                "DirectoryImage::open: path is not a directory: {}",
43                path.display()
44            )));
45        }
46        Ok(Box::new(OpenDirectory::new(path.to_path_buf())))
47    }
48
49    fn mount(&self, path: &Path) -> Result<Box<dyn MountHandle>> {
50        Ok(Box::new(SimpleMountHandle::new(path.to_path_buf())))
51    }
52
53    fn pack(&self, source_dir: &Path, output_path: &Path) -> Result<()> {
54        if source_dir == output_path {
55            return Ok(()); // no-op if source and output are the same
56        }
57        if output_path.exists() {
58            std::fs::remove_dir_all(output_path)
59                .map_err(|e| crate::Error::Format(e.to_string()))?;
60        }
61        std::fs::create_dir_all(output_path).map_err(|e| crate::Error::Format(e.to_string()))?;
62        for entry in WalkDir::new(source_dir).min_depth(1) {
63            let entry = entry.map_err(|e| crate::Error::Format(e.to_string()))?;
64            let rel = entry
65                .path()
66                .strip_prefix(source_dir)
67                .map_err(|e| crate::Error::Format(e.to_string()))?;
68            let dest = output_path.join(rel);
69            if entry.file_type().is_dir() {
70                std::fs::create_dir_all(&dest).map_err(|e| crate::Error::Format(e.to_string()))?;
71            } else {
72                if let Some(parent) = dest.parent() {
73                    std::fs::create_dir_all(parent)
74                        .map_err(|e| crate::Error::Format(e.to_string()))?;
75                }
76                std::fs::copy(entry.path(), &dest)
77                    .map_err(|e| crate::Error::Format(e.to_string()))?;
78            }
79        }
80        Ok(())
81    }
82
83    /// Create a new, writable directory image at `path`.
84    ///
85    /// Creates the directory (and any parent directories) and returns an
86    /// [`OpenDirectory`] whose [`create_partition`][crate::formats::OpenImage::create_partition]
87    /// writes partition data as files into that directory.
88    fn create(&self, path: &Path, _layout: &DiskLayout) -> Result<Box<dyn OpenImage>> {
89        std::fs::create_dir_all(path)
90            .map_err(|e| crate::Error::Format(format!("create_dir_all {}: {e}", path.display())))?;
91        Ok(Box::new(OpenDirectory::new(path.to_path_buf())))
92    }
93}

97/// [`OpenImage`] returned by `DirectoryImage::open()`.
98///
99/// The directory itself is the single filesystem partition; no actual mounting
100/// is performed.
101pub struct OpenDirectory {
102    path: PathBuf,
103    layout: DiskLayout,
104}
105
106impl OpenDirectory {
107    fn new(path: PathBuf) -> Self {
108        let layout = DiskLayout {
109            scheme: DiskScheme::SingleFs,
110            disk_guid: None,
111            partitions: vec![],
112        };
113        Self { path, layout }
114    }
115}
116
117impl OpenImage for OpenDirectory {
118    fn disk_layout(&self) -> &DiskLayout {
119        &self.layout
120    }
121
122    fn partitions(&self) -> crate::Result<Vec<PartitionHandle>> {
123        use crate::partitions::PartitionKind;
124
125        let descriptor = PartitionDescriptor {
126            number: 1,
127            partition_guid: None,
128            type_guid: None,
129            name: Some("root".into()),
130            start_lba: 0,
131            end_lba: 0,
132            size_bytes: 0,
133            flags: 0,
134            kind: PartitionKind::Fs {
135                fs_type: "directory".into(),
136            },
137        };
138
139        let path = self.path.clone();
140        let handle = FsHandle::new(descriptor, move || {
141            Ok(Box::new(SimpleMountHandle::new(path.clone())))
142        });
143
144        Ok(vec![PartitionHandle::Fs(handle)])
145    }
146
147    /// Create a writable partition handle that writes into this directory.
148    ///
149    /// - `Fs`                        → [`SimpleMountHandle`] wrapping the directory root.
150    /// - others (BiosBoot, Raw, Mbr) → error, since they don't make sense for a directory.
151    fn create_partition(&self, pm: &PartitionManifest) -> crate::Result<PartitionHandle> {
152        let desc = pm.descriptor.clone();
153        match &pm.content {
154            PartitionContent::Fs { .. } => {
155                let path = self.path.clone();
156                Ok(PartitionHandle::Fs(FsHandle::new(desc, move || {
157                    Ok(Box::new(SimpleMountHandle::new(path.clone())) as Box<dyn MountHandle>)
158                })))
159            }
160            _ => Err(crate::Error::Format(format!(
161                "OpenDirectory can only create Fs partitions, got kind {:?}",
162                desc.kind
163            ))),
164        }
165    }
166}
